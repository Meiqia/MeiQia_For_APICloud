//创建美洽
var mq = api.require('meiQia');

//配置初始化美洽需要的appkey
var param = {
    appkey:"填写您的Appkey"
};

//初始化美洽
mq.initMeiQia(param, function (ret, err){
    if (ret) {
        //alert("初始化成功：\n\n" + JSON.stringify(ret) + "\n\n点击按钮继续");
        show();
    }else{
        alert("初始化失败：\n\n" + JSON.stringify(err));
    }
})
